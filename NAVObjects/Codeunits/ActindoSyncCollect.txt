    PROCEDURE CollectRelevantEntriesForSyncBuffer@5091701();
    VAR
      Item@5091700 : Record 27;
      ItemVariant@5091701 : Record 5401;
      ItemCrossReference@5091702 : Record 5717;
      ItemTranslation@5091703 : Record 30;
      Customer@5091704 : Record 18;
    BEGIN
      //Alle Items und potenzielle Varianten durchgehen, die fÅr POS aktiviert sind
      Item.SETRANGE(POS, TRUE);
      Item.SETFILTER("Item Status", '<>%1&<>%2', Item."Item Status"::Blocked, Item."Item Status"::Sold);
      IF Item.FINDSET() THEN BEGIN
        REPEAT
          ItemVariant.SETRANGE("Item No.", Item."No.");
          ItemVariant.SETRANGE("gevis POS Active", TRUE);
          IF ItemVariant.FINDSET() THEN BEGIN
            REPEAT
              CheckIfItemMustBeUpdated(Item, ItemVariant);
            UNTIL ItemVariant.NEXT = 0;
          END ELSE BEGIN
            CLEAR(ItemVariant);
            CheckIfItemMustBeUpdated(Item, ItemVariant);
          END;
        UNTIL Item.NEXT() = 0;
      END;

      //Kunden durchsuchen
      Customer.SETRANGE(Type, Customer.Type::Person);
      Customer.SETRANGE(Blocked, Customer.Blocked::" ");
      Customer.SETFILTER("Block Reason", '');
      Customer.SETFILTER("Telex No.", '<>%1', '');
      IF Customer.FINDSET() THEN BEGIN
        REPEAT
          CheckIfCustomerMustBeUpdated(Customer);
        UNTIL Customer.NEXT() = 0;
      END;
    END;

    LOCAL PROCEDURE CheckIfItemMustBeUpdated@5091703(Item@5091700 : Record 27;ItemVariant@5091701 : Record 5401);

      ItemCrossReference@5091703 : Record 5717;

      JProperty@5091707 : DotNet "'Newtonsoft.Json, Version=6.0.0.0, Culture=neutral, PublicKeyToken=30ad4fe6b2a6aeed'.Newtonsoft.Json.Linq.JProperty";
      UpdatePriceNecessary@5091709 : Boolean;
      UpdateInventoryNecessary@5091710 : Boolean;
    BEGIN
      //Wenn noch kein Eintrag vorhanden, einen leeren anlegen
      IF NOT S04ActindoItemSyncState.GET(Item."No.", ItemVariant.Code, S04ActindoItemSyncState."Entry Type"::Item) THEN BEGIN
        S04ActindoItemSyncState.INIT;
        S04ActindoItemSyncState."Item No." := Item."No.";
        S04ActindoItemSyncState."Variant Code" := ItemVariant.Code;
        S04ActindoItemSyncState."Entry Type" := S04ActindoItemSyncState."Entry Type"::Item;
        S04ActindoItemSyncState.INSERT;
        ProcessUpdate(S04ActindoItemSyncState, 'full');
        EXIT;
      END;

      //Die restlichen Sachen erst prÅfen wenn der Artikel von Actindo bestÑtigt wurde und wir eine ID dafÅr erhalten haben
      IF Item."Actindo POS ID" <> '' THEN BEGIN

        //Item Descriptions checken
        IF S04ActindoItemSyncState.ItemDescription <> Item.Description THEN BEGIN
          S04ActindoItemSyncState.ItemDescription := Item.Description;
          FullUpdateNecessary := TRUE;
        END;
        IF S04ActindoItemSyncState.ItemDescription2 <> Item."Description 2" THEN BEGIN
          S04ActindoItemSyncState.ItemDescription2 := Item."Description 2";
          FullUpdateNecessary := TRUE;
        END;

        //EAN Codes checken
        ItemCrossReference.SETRANGE("Item No.", Item."No.");
        ItemCrossReference.SETRANGE("Cross-Reference Type", ItemCrossReference."Cross-Reference Type"::"Bar Code");
        IF ItemVariant.Code <> '' THEN
          ItemCrossReference.SETRANGE("Variant Code", ItemVariant.Code);

        IF ItemCrossReference.FINDFIRST THEN BEGIN
          IF S04ActindoItemSyncState."Cross-Reference No." <> ItemCrossReference."Cross-Reference No." THEN BEGIN
            S04ActindoItemSyncState."Cross-Reference No." := ItemCrossReference."Cross-Reference No.";
            FullUpdateNecessary := TRUE;
          END;
        END;

        //Translations checken
        CheckTranslations(S04ActindoItemSyncState, 'DEU', FullUpdateNecessary);
        CheckTranslations(S04ActindoItemSyncState, 'ENG', FullUpdateNecessary);

        IF FullUpdateNecessary THEN BEGIN
          ProcessUpdate(S04ActindoItemSyncState, 'full');
          EXIT;
        END;

        InventoryJSONObject := InventoryJSONObject.JObject();
        PricesJSONObject := PricesJSONObject.JObject();

        //Inventory Checken
        CheckInventory(S04ActindoItemSyncState, InventoryJSONObject, Item, ItemVariant.Code, UpdateInventoryNecessary);

        //Preise checken
        CheckPrice(S04ActindoItemSyncState, PricesJSONObject, Item, ItemVariant.Code, UpdatePriceNecessary);


        IF UpdateInventoryNecessary AND (NOT UpdatePriceNecessary) THEN
          ProcessUpdate(S04ActindoItemSyncState, 'inventory');

        IF UpdatePriceNecessary AND (NOT UpdateInventoryNecessary) THEN
          ProcessUpdate(S04ActindoItemSyncState, 'price');

      END;
    END;

    BEGIN
      //Sammelt BestÑnde in einem JSON
      Location.RESET();
      Location.SETFILTER("Actindo POS Location Name", '<>%1', '');
      IF Location.FINDSET() THEN BEGIN
        inventoryArray := inventoryArray.JArray();

        REPEAT
          IF Location."Actindo POS Stock Synch" THEN BEGIN
            IF VarCode <> '' THEN
              itemStockValue := S04ItemServices.GetItemStock(Item."No.", VarCode, Location.Code)
            ELSE
              itemStockValue := S04ItemServices.GetItemStock(Item."No.", '', Location.Code);

            inventoryObj := inventoryObj.JObject();

            inventoryObj.Add(prop.JProperty('compartmentId', 111)); // default value

        //Das letzte JSON mit dem aktuellen vergleichen
        S04ActindoItemSyncState.CALCFIELDS("Item Inventory");
        S04ActindoItemSyncState."Item Inventory".CREATEINSTREAM(InStream);
        InStream.READ(OldInventoryJSON);
        NewInventoryJSON := inventoryObject.ToString();
        //Update ist notwendig, JSON stimmt nicht Åberein
        IF OldInventoryJSON <> NewInventoryJSON THEN BEGIN
          S04ActindoItemSyncState."Item Inventory".CREATEOUTSTREAM(OutStream);
          OutStream.WRITE(NewInventoryJSON);
          UpdateNecessary := TRUE;
        END;

      END;

          customerObj.Add(prop.JProperty('id', customerId));
        END;

      // customer
      customerObj.Add(prop.JProperty('shortName', recDebitor."First Name" + ' ' + recDebitor.Surname));

      // check for relevant customer price group and determine Actindo customer type (employee/member)
      IF recDebitor."Telex No." <> '' THEN BEGIN
        customerObj.Add(prop.JProperty('_customer_customertyp', ActindoSetup."String Value Employee Price"));
      END ELSE BEGIN
        IF recDebitor."Customer Price Group" <> '' THEN BEGIN
          CustomerPriceGroup.GET(recDebitor."Customer Price Group");
          IF CustomerPriceGroup."Actindo POS Synch." THEN BEGIN
            CustomerPriceGroup.TESTFIELD("Actindo POS String Value");
            customerObj.Add(prop.JProperty('_customer_customertyp', CustomerPriceGroup."Actindo POS String Value"));
          END;
        END;
      END;

      customerObj.Add(prop.JProperty('_customer_debitorennumber', debNo));

      // primaryAddress
      addressObj.Add(prop.JProperty('salutation', recDebitor."Salutation Code"));
      addressObj.Add(prop.JProperty('company', recDebitor."Company Name"));
      addressObj.Add(prop.JProperty('name', recDebitor.Surname));

      addressObj.Add(prop.JProperty('phone', recDebitor."Phone No."));
      addressObj.Add(prop.JProperty('currency', 'EUR'));

      // add to payload

      debObject.Add(prop.JProperty('primaryAddress', addressObj));

        OutStream.WRITE(NewJSON);
        UpdateNecessary := TRUE;
      END;
